\chapter{Der Satz von Stokes}

\begin{thm}[Satz von Stokes]\autolabel
	Sei $M$ eine orientierte, differenzierbare Mannigfaltigkeit mit Rand, $\dim M = n$. Sei $\omega$ eine kompakt getragene (glatte) ($n-1$)-Form auf $M$. Dann gilt
	\[ \int_M d\omega = \int_{\partial M} \omega. \]
	Das $\omega$ auf der rechten Seite steht hierbei eigentlich fÃ¼r $\iota_{\partial M}^*\omega$ und $\partial M$ ist mit der kanonischen induzierten Orientierung versehen. Im Fall $n=1$ ist die rechte Seite einfach eine Summe.\\
	Ist $\partial M = \emptyset$ (also $M$ ohne Rand), so gilt
	\[ \int_M d\omega = 0. \]
\end{thm}
